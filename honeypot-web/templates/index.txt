  <script>
    let countriesChart;
    let ipsChart;
    let usersChart;
    let passwordsChart;
    let timelineChart;
    let regionReplayChart;
    let timelineRegions = [];
    let replayTimer = null;

    function createBarChart(ctx, label, labels, values, color) {
      return new Chart(ctx, {
        type: "bar",
        data: {
          labels: labels,
          datasets: [{
            label: label,
            data: values,
            backgroundColor: color
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: { display: false }
          },
          scales: {
            x: {
              ticks: { color: "#dbe6ff" },
              grid: { color: "#24314f" }
            },
            y: {
              beginAtZero: true,
              ticks: { color: "#dbe6ff" },
              grid: { color: "#24314f" }
            }
          }
        }
      });
    }

    function updateChart(chart, labels, values) {
      chart.data.labels = labels;
      chart.data.datasets[0].data = values;
      chart.update();
    }

    function createTimelineChart(ctx, labels, values) {
      return new Chart(ctx, {
        type: "line",
        data: {
          labels: labels,
          datasets: [{
            label: "Attacks per hour",
            data: values,
            borderColor: "rgba(255, 206, 86, 1)",
            backgroundColor: "rgba(255, 206, 86, 0.2)",
            fill: true,
            tension: 0.3,
            pointRadius: 3
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: { display: false }
          },
          scales: {
            x: {
              ticks: { color: "#dbe6ff", maxRotation: 0 },
              grid: { color: "#24314f" }
            },
            y: {
              beginAtZero: true,
              ticks: { color: "#dbe6ff" },
              grid: { color: "#24314f" }
            }
          }
        }
      });
    }

    function createRegionReplayChart(ctx, labels, values) {
      return new Chart(ctx, {
        type: "bar",
        data: {
          labels: labels,
          datasets: [{
            label: "Attacks by country",
            data: values,
            backgroundColor: "rgba(153, 102, 255, 0.7)"
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: { display: false }
          },
          scales: {
            x: {
              ticks: { color: "#dbe6ff" },
              grid: { color: "#24314f" }
            },
            y: {
              beginAtZero: true,
              ticks: { color: "#dbe6ff" },
              grid: { color: "#24314f" }
            }
          }
        }
      });
    }

    function renderReplayFrame(index) {
      if (!timelineRegions.length) {
        return;
      }

      const safeIndex = Math.max(0, Math.min(index, timelineRegions.length - 1));
      const bucket = timelineRegions[safeIndex];
      const regions = bucket.regions || {};
      const labels = Object.keys(regions);
      const values = Object.values(regions);
      const hourLabel = (bucket.hour || "").slice(0, 13) + ":00";

      document.getElementById("timelineSlider").value = safeIndex;
      document.getElementById("replayHourLabel").textContent = `Hour: ${hourLabel}`;

      if (!regionReplayChart) {
        regionReplayChart = createRegionReplayChart(
          document.getElementById("regionReplayChart"),
          labels,
          values
        );
      } else {
        updateChart(regionReplayChart, labels, values);
      }
    }

    function stopReplay() {
      if (replayTimer) {
        clearInterval(replayTimer);
        replayTimer = null;
      }
    }

    function startReplay() {
      if (!timelineRegions.length || replayTimer) {
        return;
      }
      replayTimer = setInterval(() => {
        const slider = document.getElementById("timelineSlider");
        let next = Number(slider.value) + 1;
        if (next >= timelineRegions.length) {
          next = 0;
        }
        renderReplayFrame(next);
      }, 1000);
    }

    async function fetchAlerts() {
      try {
        const response = await fetch("/alerts-data");
        const alerts = await response.json();
        const alertsList = document.getElementById("alertsList");

        if (!alerts.length) {
          alertsList.innerHTML = '<li class="list-group-item">No alerts available.</li>';
          return;
        }

        alertsList.innerHTML = alerts.map((alert) => {
          const ip = alert.ip || "Unknown";
          const country = alert.country || "Unknown";
          const city = alert.city || "Unknown";
          const user = alert.username || "";
          const userChunk = user ? ` | user: ${user}` : "";
          return `<li class="list-group-item"><strong>${ip}</strong> - ${country}, ${city}${userChunk}</li>`;
        }).join("");
      } catch (error) {
        console.error("Failed to fetch alerts:", error);
      }
    }

    async function fetchStats() {
      try {
        const response = await fetch("/stats");
        const stats = await response.json();

        const countryLabels = Object.keys(stats.country_counts || {});
        const countryValues = Object.values(stats.country_counts || {});
        const ipLabels = Object.keys(stats.ip_counts || {});
        const ipValues = Object.values(stats.ip_counts || {});
        const userLabels = Object.keys(stats.username_counts || {});
        const userValues = Object.values(stats.username_counts || {});
        const passwordLabels = Object.keys(stats.password_counts || {});
        const passwordValues = Object.values(stats.password_counts || {});
        const counter = document.getElementById("attackCounter");
        counter.textContent = stats.total_attacks || 0;
	      document.getElementById("attacksLast5Minutes").textContent =
	          stats.attacks_last_5_minutes || 0;

        if (!countriesChart) {
          countriesChart = createBarChart(
            document.getElementById("countriesChart"),
            "Countries",
            countryLabels,
            countryValues,
            "rgba(54, 162, 235, 0.7)"
          );
          ipsChart = createBarChart(
            document.getElementById("ipsChart"),
            "IPs",
            ipLabels,
            ipValues,
            "rgba(255, 99, 132, 0.7)"
          );
          usersChart = createBarChart(
            document.getElementById("usersChart"),
            "Usernames",
            userLabels,
            userValues,
            "rgba(75, 192, 192, 0.7)"
          );
          passwordsChart = createBarChart(
            document.getElementById("passwordsChart"),
            "Passwords",
            passwordLabels,
            passwordValues,
            "rgba(255, 159, 64, 0.8)"
          );
        } else {
          updateChart(countriesChart, countryLabels, countryValues);
          updateChart(ipsChart, ipLabels, ipValues);
          updateChart(usersChart, userLabels, userValues);
          updateChart(passwordsChart, passwordLabels, passwordValues);
        }
      } catch (error) {
        console.error("Failed to fetch stats:", error);
      }
    }

    async function fetchTimeline() {
      try {
        const response = await fetch("/timeline");
        const points = await response.json();
        const labels = points.map((p) => (p.hour || "").slice(11, 16));
        const values = points.map((p) => p.count || 0);

        if (!timelineChart) {
          timelineChart = createTimelineChart(
            document.getElementById("timelineChart"),
            labels,
            values
          );
        } else {
          updateChart(timelineChart, labels, values);
        }
      } catch (error) {
        console.error("Failed to fetch timeline:", error);
      }
    }

    async function fetchTimelineRegions() {
      try {
        const response = await fetch("/timeline-regions");
        timelineRegions = await response.json();
        const slider = document.getElementById("timelineSlider");

        if (!timelineRegions.length) {
          slider.min = 0;
          slider.max = 0;
          slider.value = 0;
          document.getElementById("replayHourLabel").textContent = "Hour: --";
          return;
        }

        slider.min = 0;
        slider.max = String(timelineRegions.length - 1);

        const current = Number(slider.value || 0);
        renderReplayFrame(Math.min(current, timelineRegions.length - 1));
      } catch (error) {
        console.error("Failed to fetch timeline regions:", error);
      }
    }

    document.getElementById("timelineSlider").addEventListener("input", (event) => {
      stopReplay();
      renderReplayFrame(Number(event.target.value));
    });
    document.getElementById("playReplayBtn").addEventListener("click", startReplay);
    document.getElementById("pauseReplayBtn").addEventListener("click", stopReplay);

    fetchAlerts();
    fetchStats();
    fetchTimeline();
    fetchTimelineRegions();

    setInterval(fetchAlerts, 5000);
    setInterval(fetchStats, 10000);
    setInterval(fetchTimeline, 10000);
    setInterval(fetchTimelineRegions, 10000);
  </script>
